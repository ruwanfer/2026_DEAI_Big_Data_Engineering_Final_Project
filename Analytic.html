import requests
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

BASE = "http://localhost:8000"

# ── Q1: Peak hour demand ──────────────────────────────────────────────────────
r = requests.get(f"{BASE}/analysis/peak-hours")
df = pd.DataFrame(r.json()["data"]["Uber"]["peak_hours_top5"] +
                  r.json()["data"]["Lyft"]["peak_hours_top5"])

plt.figure(figsize=(12, 5))
for company in df["company"].unique():
    sub = df[df["company"] == company]
    plt.plot(sub["pickup_hour"], sub["total_rides"], marker="o", label=company)
plt.title("Q1: Ride Demand by Hour of Day")
plt.xlabel("Hour")
plt.ylabel("Total Rides")
plt.legend()
plt.tight_layout()
plt.savefig("q1_peak_hours.png")
plt.show()

# ── Q6: Fare by hour ──────────────────────────────────────────────────────────
r = requests.get(f"{BASE}/fare-stats")
df = pd.DataFrame(r.json()["data"])

plt.figure(figsize=(12, 5))
for company in df["company"].unique():
    sub = df[df["company"] == company].groupby("pickup_hour")["avg_fare"].mean()
    plt.plot(sub.index, sub.values, marker="o", label=company)
plt.title("Q6: Average Fare by Hour of Day")
plt.xlabel("Hour")
plt.ylabel("Avg Fare ($)")
plt.legend()
plt.tight_layout()
plt.savefig("q6_fare_by_hour.png")
plt.show()
